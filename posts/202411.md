```python
import math

# 给定方程 A^2 + B^2 = 5445
target = 5445

# 尝试不同的 A 值
for A in range(1, int(math.sqrt(target)) + 1):
    B_squared = target - A**2
    B = int(math.sqrt(B_squared))
    if B**2 == B_squared:  # 如果 B^2 是完全平方数
        print(f"A = {A}, B = {B}")
```
